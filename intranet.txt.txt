<script>
document.addEventListener("DOMContentLoaded", function() {
var el = document.getElementById("modal-one");
  el.classList.add("open");   
document.querySelector('#modal-exit').addEventListener('click', function (e) {
e.preventDefault();
  var el = document.getElementById("modal-one");
  el.classList.remove("open");    
});
});
</script>
